create table public.images (
  id text not null,
  dataset_id text null,
  filename text not null,
  storage_path text not null,
  width integer null,
  height integer null,
  uploaded_at timestamp with time zone null default now(),
  user_id uuid null,
  image_type text not null default 'raw'::text,
  parent_image_id text null,
  constraint images_pkey primary key (id),
  constraint images_parent_image_id_fkey foreign KEY (parent_image_id) references images (id),
  constraint images_image_type_check check (
    (
      image_type = any (array['raw'::text, 'slip'::text, 'char'::text])
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_images_user_type on public.images using btree (user_id, image_type) TABLESPACE pg_default;

create index IF not exists idx_images_parent on public.images using btree (parent_image_id) TABLESPACE pg_default;


create table public.segment_metadata (
  id uuid not null default gen_random_uuid (),
  segment_id uuid not null,
  title text null,
  content_description text null,
  event_type text null,
  event_date date null,
  place text null,
  people jsonb null,
  extra jsonb null,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  constraint segment_metadata_pkey primary key (id),
  constraint segment_metadata_segment_id_fkey foreign KEY (segment_id) references segments (id) on delete CASCADE
) TABLESPACE pg_default;

create index IF not exists idx_segment_metadata_segment_id on public.segment_metadata using btree (segment_id) TABLESPACE pg_default;
